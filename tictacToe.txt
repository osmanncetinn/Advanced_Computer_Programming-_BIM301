    private void winning_game(){
        String b1=jButton1.getText();
         String b2=jButton1.getText();
          String b3=jButton1.getText();
           String b4=jButton1.getText();
            String b5=jButton1.getText();
             String b6=jButton1.getText();
              String b7=jButton1.getText();
               String b8=jButton1.getText();
                String b9=jButton1.getText();
                if(b1==("X")&&b2==("X")&&b3==("X")){
                    JOptionPane.showMessageDialog(this,"Player Xwin", "TicTacToe", JOptionPane.INFORMATION_MESSAGE);
                    xCount++;
                    gameScore();
                    jButton1.setBackground(Color.YELLOW);
                    jButton2.setBackground(Color.YELLOW);
                    jButton3.setBackground(Color.YELLOW);
                    
                }
                if(b4==("X")&&b5==("X")&&b6==("X")){
                    JOptionPane.showMessageDialog(this,"Player Xwin", "TicTacToe", JOptionPane.INFORMATION_MESSAGE);
                    xCount++;
                    gameScore();
                    jButton4.setBackground(Color.BLUE);
                    jButton5.setBackground(Color.BLUE);
                    jButton6.setBackground(Color.BLUE);
